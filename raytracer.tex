\documentclass{standalone}
\usepackage{color}
\usepackage{pgffor}
\usepackage{xfp}

\newcommand{\dimension}{1pt}
\newcommand{\pixel}[1]{\color[RGB]{#1}{\rule{\dimension}{\dimension}}}

\newcommand{\width}{100}
\newcommand{\height}{100}

% #1 = x, #2 = y
\def\createpixel#1#2{%
    \pixel{
        \fpeval{floor(255.0 * (#1 / (\width - 1)))},
        \fpeval{floor(255.0 * (#2 / (\height - 1)))},
        \fpeval{floor(255.0 * 0.25)}
    }
}

\begin{document}

\unitlength=\dimension
\begin{picture}(\width,\height)
    \foreach \x in {0,...,\inteval{\width - 1}}{
        \foreach \y in {0,...,\inteval{\height - 1}}{
            \put(\x,\y){\createpixel{\x}{\y}}
        }
    }
\end{picture}

\end{document}
